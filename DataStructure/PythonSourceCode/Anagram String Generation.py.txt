def generate_anagram(s):
    if len(s) <=1:
        return s
    else:
        res = []
        for perm in generate_anagram(s[1:]):
            for i in range(len(s)):
                res.append(perm[:i] + s[0:1] + perm[i:])
        return set(res)

if __name__ == "__main__":
    print(generate_anagram("anagram"))
